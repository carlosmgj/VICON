118 ---------------------------------------------------------------------------

120 ---------------------------------------------------------------------------

122 ST_IDLE,
123 -- START / Repeated START

202 ---------------------------------------------------------------------------
203 -- WR FIFO
204 -- wr_dout es combinacional sobre wr_rd_ptr (lectura asíncrona).
205 -- El pop avanza wr_rd_ptr en el flanco siguiente, por eso se necesita
206 -- ST_LOAD_DATA_H para capturar wr_word antes de usarlo en TX.
207 ---------------------------------------------------------------------------
208 wr_full_i <= '1' when wr_count = FIFO_DEPTH else '0';
209 wr_empty_i <= '1' when wr_count = 0 else '0';
210 wr_fifo_full <= wr_full_i;
211 wr_fifo_empty <= wr_empty_i;
212 wr_dout <= wr_mem(wr_rd_ptr);
213
214 p_wr_fifo : process(clk)
215 variable do_push : boolean;
216 variable do_pop : boolean;
217 begin
218 if rising_edge(clk) then
219 if reset = '1' then
220 wr_wr_ptr <= 0; wr_rd_ptr <= 0; wr_count <= 0;
221 else
222 do_push := (wr_fifo_push = '1') and (wr_full_i = '0');
223 do_pop := (wr_pop = '1') and (wr_empty_i = '0');
224 if do_push then
225 wr_mem(wr_wr_ptr) <= wr_fifo_data;
226 wr_wr_ptr <= (wr_wr_ptr + 1) mod FIFO_DEPTH;
227 end if;
228 if do_pop then
229 wr_rd_ptr <= (wr_rd_ptr + 1) mod FIFO_DEPTH;
230 end if;
231 if do_push and not do_pop then wr_count <= wr_count + 1;
232 elsif do_pop and not do_push then wr_count <= wr_count - 1;
233 end if;
234 end if;
235 end if;
236 end process p_wr_fifo;
237
238 ---------------------------------------------------------------------------
239 -- RD FIFO
240 ---------------------------------------------------------------------------
241 rd_full_i <= '1' when rd_count = FIFO_DEPTH else '0';
242 rd_empty_i <= '1' when rd_count = 0 else '0';
243 rd_fifo_full <= rd_full_i;
244 rd_fifo_empty <= rd_empty_i;
245 rd_fifo_data <= rd_mem(rd_rd_ptr);
246
247 p_rd_fifo : process(clk)
248 variable do_push : boolean;
249 variable do_pop : boolean;
250 begin
251 if rising_edge(clk) then
252 if reset = '1' then
253 rd_wr_ptr <= 0; rd_rd_ptr <= 0; rd_count <= 0;
254 else
255 do_push := (rd_push = '1') and (rd_full_i = '0');
256 do_pop := (rd_fifo_pop = '1') and (rd_empty_i = '0');
257 if do_push then
258 rd_mem(rd_wr_ptr) <= rd_din;
259 rd_wr_ptr <= (rd_wr_ptr + 1) mod FIFO_DEPTH;
260 end if;
261 if do_pop then
262 rd_rd_ptr <= (rd_rd_ptr + 1) mod FIFO_DEPTH;
263 end if;
264 if do_push and not do_pop then rd_count <= rd_count + 1;
265 elsif do_pop and not do_push then rd_count <= rd_count - 1;
266 end if;
267 end if;
268 end if;
269 end process p_rd_fifo;
270
271 ---------------------------------------------------------------------------
272 -- Generador de fase
273 ---------------------------------------------------------------------------
274 p_phase : process(clk)
275 begin
276 if rising_edge(clk) then
277 if reset = '1' then
278 phase_cnt <= 0; phase_tick <= '0';
279 elsif phase_cnt = CLKS_PER_PHASE - 1 then
280 phase_cnt <= 0; phase_tick <= '1';
281 else
282 phase_cnt <= phase_cnt + 1;
283 phase_tick <= '0';
284 end if;
285 end if;
286 end process p_phase;
287
288 ---------------------------------------------------------------------------
289 -- FSM principal
290 ---------------------------------------------------------------------------
291 p_fsm : process(clk)
292 begin
293 if rising_edge(clk) then
294 if reset = '1' then
295 state <= ST_IDLE;
296 seq_next <= ST_IDLE;
297 busy_i <= '0';
298 done_i <= '0';
299 error_i <= '0';
300 scl_r <= '1';
301 sda_out_r <= '1';
302 sda_oe <= '1';
303 wr_pop <= '0';
304 rd_push <= '0';
305 rd_din <= (others => '0');
306 tx_byte <= (others => '0');
307 rx_byte <= (others => '0');
308 rx_high <= (others => '0');
309 wr_word <= (others => '0');
310 bit_cnt <= 7;
311 reg_cnt <= 0;
312 send_nack <= '0';
313 start_rd_mode <= '0';
314 r_rw <= '0';
315 r_addr_dev <= (others => '0');
316 r_addr_reg <= (others => '0');
317 r_num_regs <= 1;
318 else
319 wr_pop <= '0';
320 rd_push <= '0';
321 done_i <= '0';
322
323 case state is
324
325 -----------------------------------------------------------
326 when ST_IDLE =>
327 scl_r <= '1';
328 sda_out_r <= '1';
329 sda_oe <= '1';
330 busy_i <= '0';
331 error_i <= '0';
332 if start_i2c = '1' then
333 if rd_empty_i = '0' then
334 null; -- RD FIFO no vaciada: bloquear
335 else
336 r_rw <= rw;
337 r_addr_dev <= addr_dev;
338 r_addr_reg <= addr_reg;
339 r_num_regs <= num_regs;
340 reg_cnt <= 0;
341 start_rd_mode <= '0';
342 busy_i <= '1';
343 state <= ST_START_0;
344 end if;
345 end if;
346
347 -----------------------------------------------------------
348 -- START / Repeated START
349 -- SDA baja mientras SCL está alto → condición START
350 -----------------------------------------------------------
351 when ST_START_0 =>
352 if phase_tick = '1' then
353 scl_r <= '0'; sda_out_r <= '1'; sda_oe <= '1';
354 state <= ST_START_1;
355 end if;
356
357 when ST_START_1 =>
358 if phase_tick = '1' then
359 scl_r <= '1'; state <= ST_START_2;
360 end if;
361
362 when ST_START_2 =>
363 if phase_tick = '1' then
364 sda_out_r <= '0'; -- SDA baja con SCL alto → START
365 state <= ST_START_3;
366 end if;
367
368 when ST_START_3 =>
369 if phase_tick = '1' then
370 scl_r <= '0';
371 bit_cnt <= 7;
372 if start_rd_mode = '0' then
373 tx_byte <= r_addr_dev & '0'; -- Write
374 seq_next <= ST_DECIDE_AFTER_ADDR_WR;
375 else
376 tx_byte <= r_addr_dev & '1'; -- Read
377 seq_next <= ST_DECIDE_AFTER_ADDR_RD;
378 end if;
379 state <= ST_TX_0;
380 end if;
381
382 -----------------------------------------------------------
383 -- TX byte reutilizable
384 -- Precargar: tx_byte, bit_cnt=7, seq_next
385 -----------------------------------------------------------
386 when ST_TX_0 =>
387 if phase_tick = '1' then
388 scl_r <= '0';
389 sda_oe <= '1';
390 sda_out_r <= tx_byte(bit_cnt);
391 state <= ST_TX_1;
392 end if;
393
394 when ST_TX_1 =>
395 if phase_tick = '1' then
396 scl_r <= '1'; state <= ST_TX_2;
397 end if;
398
399 when ST_TX_2 =>
400 if phase_tick = '1' then
401 state <= ST_TX_3;
402 end if;
403
404 when ST_TX_3 =>
405 if phase_tick = '1' then
406 scl_r <= '0';
407 if bit_cnt = 0 then
408 bit_cnt <= 7;
409 state <= ST_RACK_0;
410 else
411 bit_cnt <= bit_cnt - 1;
412 state <= ST_TX_0;
413 end if;
414 end if;
415
416 -----------------------------------------------------------
417 -- RACK: ACK del esclavo (SDA='0'=ACK, '1'=NACK)
418 -----------------------------------------------------------
419 when ST_RACK_0 =>
420 if phase_tick = '1' then
421 scl_r <= '0'; sda_oe <= '0'; state <= ST_RACK_1;
422 end if;
423
424 when ST_RACK_1 =>
425 if phase_tick = '1' then
426 scl_r <= '1'; state <= ST_RACK_2;
427 end if;
428
429 when ST_RACK_2 =>
430 if phase_tick = '1' then
431 if sdata = '1' then
432 state <= ST_ERROR_STOP;
433 else
434 state <= ST_RACK_3;
435 end if;
436 end if;
437
438 when ST_RACK_3 =>
439 if phase_tick = '1' then
440 scl_r <= '0'; state <= seq_next;
441 end if;
442
443 -----------------------------------------------------------
444 -- Decisión WRITE: tras RACK de ADDR_WR → enviar REG_ADDR
445 -----------------------------------------------------------
446 when ST_DECIDE_AFTER_ADDR_WR =>
447 tx_byte <= r_addr_reg;
448 bit_cnt <= 7;
449 seq_next <= ST_DECIDE_AFTER_REG_ADDR;
450 state <= ST_TX_0;
451
452 -----------------------------------------------------------
453 -- Decisión: tras RACK de REG_ADDR
454 -- Write → capturar dato de WR FIFO, esperar un ciclo
455 -- Read → Repeated START con R/W='1'
456 -----------------------------------------------------------
457 when ST_DECIDE_AFTER_REG_ADDR =>
458 if r_rw = '0' then
459 if wr_empty_i = '0' then
460 wr_word <= wr_dout; -- capturar dato (wr_dout combinacional)
461 wr_pop <= '1'; -- avanzar puntero en el flanco siguiente
462 state <= ST_LOAD_DATA_H;
463 end if;
464 -- Si FIFO vacía: esperar en este estado
465 else
466 start_rd_mode <= '1';
467 state <= ST_START_0;
468 end if;
469
470 -----------------------------------------------------------
471 -- ST_LOAD_DATA_H: wr_word ya estable → cargar byte alto en TX
472 -----------------------------------------------------------
473 when ST_LOAD_DATA_H =>
474 tx_byte <= wr_word(15 downto 8);
475 bit_cnt <= 7;
476 seq_next <= ST_DECIDE_AFTER_DATA_H;
477 state <= ST_TX_0;
478
479 -----------------------------------------------------------
480 -- Decisión: tras RACK de DATA_H → enviar byte bajo
481 -----------------------------------------------------------
482 when ST_DECIDE_AFTER_DATA_H =>
483 tx_byte <= wr_word(7 downto 0);
484 bit_cnt <= 7;
485 seq_next <= ST_DECIDE_AFTER_DATA_L;
486 state <= ST_TX_0;
487
488 -----------------------------------------------------------
489 -- Decisión: tras RACK de DATA_L → ¿más registros?
490 -----------------------------------------------------------
491 when ST_DECIDE_AFTER_DATA_L =>
492 reg_cnt <= reg_cnt + 1;
493 if reg_cnt + 1 = r_num_regs then
494 state <= ST_STOP_0; -- todos escritos
495 else
496 -- Más registros: capturar siguiente dato
497 if wr_empty_i = '0' then
498 wr_word <= wr_dout;
499 wr_pop <= '1';
500 state <= ST_LOAD_NEXT_DATA_H;
501 end if;
502 -- Si FIFO vacía: esperar aquí
503 end if;
504
505 -----------------------------------------------------------
506 -- ST_LOAD_NEXT_DATA_H: igual que ST_LOAD_DATA_H
507 -- (estado separado para claridad; mismo comportamiento)
508 -----------------------------------------------------------
509 when ST_LOAD_NEXT_DATA_H =>
510 tx_byte <= wr_word(15 downto 8);
511 bit_cnt <= 7;
512 seq_next <= ST_DECIDE_AFTER_DATA_H;
513 state <= ST_TX_0;
514
515 -----------------------------------------------------------
516 -- Decisión READ: tras RACK de ADDR_RD → recibir DATA_H
517 -----------------------------------------------------------
518 when ST_DECIDE_AFTER_ADDR_RD =>
519 send_nack <= '0';
520 bit_cnt <= 7;
521 seq_next <= ST_DECIDE_AFTER_RX_H;
522 state <= ST_RX_0;
523
524 -----------------------------------------------------------
525 -- Decisión: tras MACK de DATA_H → recibir DATA_L
526 -- Activar NACK si este es el último registro
527 -----------------------------------------------------------
528 when ST_DECIDE_AFTER_RX_H =>
529 rx_high <= rx_byte;
530 if reg_cnt + 1 = r_num_regs then
531 send_nack <= '1'; -- último byte → NACK
532 else
533 send_nack <= '0';
534 end if;
535 bit_cnt <= 7;
536 seq_next <= ST_DECIDE_AFTER_RX_L;
537 state <= ST_RX_0;
538
539 -----------------------------------------------------------
540 -- Decisión: tras MACK de DATA_L → push RD FIFO, ¿más?
541 -----------------------------------------------------------
542 when ST_DECIDE_AFTER_RX_L =>
543 if rd_full_i = '0' then
544 rd_din <= rx_high & rx_byte;
545 rd_push <= '1';
546 reg_cnt <= reg_cnt + 1;
547 if reg_cnt + 1 = r_num_regs then
548 state <= ST_STOP_0; -- todos leídos
549 else
550 send_nack <= '0';
551 bit_cnt <= 7;
552 seq_next <= ST_DECIDE_AFTER_RX_H;
553 state <= ST_RX_0;
554 end if;
555 else
556 state <= ST_ERROR_STOP; -- RD FIFO llena
557 end if;
558
559 -----------------------------------------------------------
560 -- RX byte reutilizable
561 -- Precargar: send_nack, bit_cnt=7, seq_next
562 -----------------------------------------------------------
563 when ST_RX_0 =>
564 if phase_tick = '1' then
565 scl_r <= '0'; sda_oe <= '0'; state <= ST_RX_1;
566 end if;
567
568 when ST_RX_1 =>
569 if phase_tick = '1' then
570 scl_r <= '1'; state <= ST_RX_2;
571 end if;
572
573 when ST_RX_2 =>
574 if phase_tick = '1' then
575 rx_byte(bit_cnt) <= sdata; state <= ST_RX_3;
576 end if;
577
578 when ST_RX_3 =>
579 if phase_tick = '1' then
580 scl_r <= '0';
581 if bit_cnt = 0 then
582 bit_cnt <= 7;
583 state <= ST_MACK_0;
584 else
585 bit_cnt <= bit_cnt - 1;
586 state <= ST_RX_0;
587 end if;
588 end if;
589
590 -----------------------------------------------------------
591 -- MACK: master envía ACK ('0') o NACK ('1') según send_nack
592 -----------------------------------------------------------
593 when ST_MACK_0 =>
594 if phase_tick = '1' then
595 scl_r <= '0';
596 sda_oe <= '1';
597 sda_out_r <= send_nack;
598 state <= ST_MACK_1;
599 end if;
600
601 when ST_MACK_1 =>
602 if phase_tick = '1' then
603 scl_r <= '1'; state <= ST_MACK_2;
604 end if;
605
606 when ST_MACK_2 =>
607 if phase_tick = '1' then
608 state <= ST_MACK_3;
609 end if;
610
611 when ST_MACK_3 =>
612 if phase_tick = '1' then
613 scl_r <= '0'; state <= seq_next;
614 end if;
615
616 -----------------------------------------------------------
617 -- STOP: SDA sube mientras SCL está alto
618 -----------------------------------------------------------
619 when ST_STOP_0 =>
620 if phase_tick = '1' then
621 scl_r <= '0'; sda_out_r <= '0'; sda_oe <= '1';
622 state <= ST_STOP_1;
623 end if;
624
625 when ST_STOP_1 =>
626 if phase_tick = '1' then
627 scl_r <= '1'; state <= ST_STOP_2;
628 end if;
629
630 when ST_STOP_2 =>
631 if phase_tick = '1' then
632 sda_out_r <= '1'; -- SDA sube con SCL alto → STOP
633 state <= ST_STOP_3;
634 end if;
635
636 when ST_STOP_3 =>
637 if phase_tick = '1' then
638 state <= ST_DONE;
639 end if;
640
641 -----------------------------------------------------------
642 when ST_DONE =>
643 done_i <= '1';
644 busy_i <= '0';
645 state <= ST_IDLE;
646
647 when ST_ERROR_STOP =>
648 error_i <= '1';
649 state <= ST_STOP_0; -- liberar bus antes de volver a IDLE
650
651 when ST_ERROR =>
652 error_i <= '1';
653 busy_i <= '0';
654 state <= ST_IDLE;
655
656 when others =>
657 state <= ST_IDLE;
658
659 end case;
660 end if;
661 end if;
662 end process p_fsm;
663
664 ---------------------------------------------------------------------------
665 -- Salidas
666 ---------------------------------------------------------------------------
667 sclk <= scl_r;
668 sdata <= sda_out_r when sda_oe = '1' else 'Z';
669 busy <= busy_i;
670 done <= done_i;
671 error <= error_i;
672
673end architecture rtl;
