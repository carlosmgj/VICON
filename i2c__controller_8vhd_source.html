4library IEEE;
5use IEEE.STD_LOGIC_1164.ALL;
6use IEEE.NUMERIC_STD.ALL;

8entity I2C_CONTROLLER is
9 generic (
10 CLK_DIV_G : positive := 250 -- Medio periodo I2C en ciclos de clk
11 );
12 port (
13 clk : in std_logic;
14 reset : in std_logic;
15 -- CONTROL
16 rw : in std_logic; -- '0'=Write, '1'=Read
17 start : in std_logic;
18 num_bytes: in integer range 1 to 4; -- Bytes de datos (modo write)
19 -- DATOS
20 addr_dev : in std_logic_vector(6 downto 0);
21 addr_reg : in std_logic_vector(7 downto 0);
22 data_wr : in std_logic_vector(31 downto 0); -- Datos a escribir (MSB first)
23 data_rd : out std_logic_vector(7 downto 0); -- Byte leido (modo read)
24 -- ESTADO
25 busy : out std_logic;
26 done : out std_logic;
27 error : out std_logic;
28 -- BUS I2C
29 sclk : out std_logic;
30 sdata : inout std_logic

32end I2C_CONTROLLER;

34architecture Behavioral of I2C_CONTROLLER is
35
36 -- -------------------------------------------------------------------------
37 -- Tipo de estado
38 -- -------------------------------------------------------------------------
39 type state_t is (
40 IDLE,
41 START_SDA_LOW,
42 START_SCL_LOW,
43 TX_BIT_LOW,
44 TX_BIT_HIGH,
45 ACK_SCL_LOW,
46 ACK_SCL_HIGH,
47 RX_BIT_LOW,
48 RX_BIT_HIGH,
49 MACK_SCL_LOW,
50 MACK_SCL_HIGH,
51 STOP_SCL_LOW,
52 STOP_SDA_LOW,
53 STOP_SCL_HIGH,
54 FINISHED,
55 ERROR_STATE
56 );
57
58 signal state : state_t := IDLE;
59
60 -- -------------------------------------------------------------------------
61 -- Clock divider
62 -- -------------------------------------------------------------------------
63 signal clk_cnt : integer range 0 to CLK_DIV_G-1 := 0;
64 signal tick : std_logic := '0'; -- '1' un ciclo cada CLK_DIV_G ciclos
65
66 -- -------------------------------------------------------------------------
67 -- Registros internos
68 -- -------------------------------------------------------------------------
69 signal bit_cnt : integer range 0 to 7 := 7;
70 signal byte_cnt : integer range 0 to 5 := 0; -- 0=addr+W, 1=reg, 2..=datos, last=addr+R
71
72 signal shift_reg : std_logic_vector(7 downto 0) := (others => '0');
73 signal rx_reg : std_logic_vector(7 downto 0) := (others => '0');
74
75 signal sclk_r : std_logic := '1';
76 signal sdata_out : std_logic := '1';
77 signal sdata_oe : std_logic := '0'; -- '1' = master conduce SDA
78
79 -- Registros de los parametros de la transaccion (latcheados al inicio)
80 signal rw_r : std_logic := '0';
81 signal nb_r : integer range 1 to 4 := 1;
82 signal addr_r : std_logic_vector(6 downto 0) := (others => '0');
83 signal areg_r : std_logic_vector(7 downto 0) := (others => '0');
84 signal dwr_r : std_logic_vector(31 downto 0) := (others => '0');
85
86 -- Flag para saber si el siguiente ACK es el ultimo antes de parar
87 signal last_data : std_logic := '0';
88 -- Flag para saber si estamos en la fase de lectura (tras repeated start)
89 signal in_read : std_logic := '0';
90
91begin
92
93 -- =========================================================================
94 -- Buffer triestado SDA
95 -- =========================================================================
96 sdata <= sdata_out when sdata_oe = '1' else 'Z';
97 sclk <= sclk_r;
98
99 -- =========================================================================
100 -- Generador de tick (1 pulso cada CLK_DIV_G ciclos)
101 -- =========================================================================
102 p_clkdiv : process(clk)
103 begin
104 if rising_edge(clk) then
105 if reset = '1' then
106 clk_cnt <= 0;
107 tick <= '0';
108 elsif clk_cnt = CLK_DIV_G - 1 then
109 clk_cnt <= 0;
110 tick <= '1';
111 else
112 clk_cnt <= clk_cnt + 1;
113 tick <= '0';
114 end if;
115 end if;
116 end process p_clkdiv;
117
118 -- =========================================================================
119 -- Maquina de estados - avanza solo en cada tick
120 -- =========================================================================
121 p_fsm : process(clk)
122 begin
123 if rising_edge(clk) then
124 if reset = '1' then
125 state <= IDLE;
126 sclk_r <= '1';
127 sdata_out <= '1';
128 sdata_oe <= '0';
129 busy <= '0';
130 done <= '0';
131 error <= '0';
132 bit_cnt <= 7;
133 byte_cnt <= 0;
134 in_read <= '0';
135 last_data <= '0';
136 data_rd <= (others => '0');
137
138 elsif tick = '1' then
139
140 -- Defecto: pulsos de un ciclo
141 done <= '0';
142 error <= '0';
143
144 case state is
145
146 -- ---------------------------------------------------------
147 when IDLE =>
148 busy <= '0';
149 sclk_r <= '1';
150 sdata_out <= '1';
151 sdata_oe <= '1';
152 in_read <= '0';
153 if start = '1' then
154 -- Latch
155 rw_r <= rw;
156 nb_r <= num_bytes;
157 addr_r <= addr_dev;
158 areg_r <= addr_reg;
159 dwr_r <= data_wr;
160 busy <= '1';
161 byte_cnt <= 0;
162 state <= START_SDA_LOW;
163 end if;
164
165 -- ---------------------------------------------------------
166 -- START: SDA baja con SCL alto
167 -- ---------------------------------------------------------
168 when START_SDA_LOW =>
169 sdata_out <= '0'; -- SDA baja ? START
170 state <= START_SCL_LOW;
171
172 when START_SCL_LOW =>
173 sclk_r <= '0'; -- SCL baja para empezar a transmitir
174 -- Cargamos primer byte: ADDR + R/W
175 -- Si in_read='1' estamos en repeated start ? ADDR+R
176 -- Si in_read='0' ? ADDR+W
177 if in_read = '1' then
178 shift_reg <= addr_r & '1';
179 else
180 shift_reg <= addr_r & '0';
181 end if;
182 bit_cnt <= 7;
183 state <= TX_BIT_LOW;
184
185 -- ---------------------------------------------------------
186 -- Transmision de un byte (master ? esclavo)
187 -- ---------------------------------------------------------
188 when TX_BIT_LOW =>
189 sclk_r <= '0';
190 sdata_oe <= '1';
191 sdata_out <= shift_reg(bit_cnt);
192 state <= TX_BIT_HIGH;
193
194 when TX_BIT_HIGH =>
195 sclk_r <= '1';
196 if bit_cnt = 0 then
197 state <= ACK_SCL_LOW;
198 else
199 bit_cnt <= bit_cnt - 1;
200 state <= TX_BIT_LOW;
201 end if;
202
203 -- ---------------------------------------------------------
204 -- Espera ACK del esclavo
205 -- ---------------------------------------------------------
206 when ACK_SCL_LOW =>
207 sclk_r <= '0';
208 sdata_oe <= '0'; -- Soltamos SDA para que esclavo ponga ACK
209 state <= ACK_SCL_HIGH;
210
211 when ACK_SCL_HIGH =>
212 sclk_r <= '1';
213 if sdata = '1' then
214 -- NACK ? abortar
215 state <= ERROR_STATE;
216 else
217 -- ACK OK ? decidir siguiente accion segun byte_cnt
218 if in_read = '1' then
219 -- Acabamos de enviar ADDR+R, empezamos a leer
220 bit_cnt <= 7;
221 rx_reg <= (others => '0');
222 state <= RX_BIT_LOW;
223 elsif byte_cnt = 0 then
224 -- Fin de ADDR+W ? enviar registro
225 shift_reg <= areg_r;
226 bit_cnt <= 7;
227 byte_cnt <= 1;
228 state <= TX_BIT_LOW;
229 elsif byte_cnt = 1 then
230 -- Fin de REG
231 if rw_r = '1' then
232 -- Repeated START para lectura
233 in_read <= '1';
234 sdata_out <= '1';
235 sdata_oe <= '1';
236 state <= START_SDA_LOW;
237 else
238 -- Escritura: enviar bytes de datos
239 shift_reg <= dwr_r(31 downto 24);
240 bit_cnt <= 7;
241 byte_cnt <= 2;
242 state <= TX_BIT_LOW;
243 end if;
244 elsif byte_cnt <= nb_r then
245 -- Enviando bytes de datos (byte_cnt va de 2 a nb_r+1)
246 -- byte_cnt=2 ? dwr_r(23..16), 3?(15..8), 4?(7..0)
247 case byte_cnt is
248 when 2 => shift_reg <= dwr_r(23 downto 16);
249 when 3 => shift_reg <= dwr_r(15 downto 8);
250 when others => shift_reg <= dwr_r(7 downto 0);
251 end case;
252 bit_cnt <= 7;
253 byte_cnt <= byte_cnt + 1;
254 state <= TX_BIT_LOW;
255 else
256 -- Todos los bytes enviados ? STOP
257 state <= STOP_SCL_LOW;
258 end if;
259 end if;
260
261 -- ---------------------------------------------------------
262 -- Recepcion de un byte (esclavo ? master)
263 -- ---------------------------------------------------------
264 when RX_BIT_LOW =>
265 sclk_r <= '0';
266 sdata_oe <= '0'; -- Master en alta impedancia
267 state <= RX_BIT_HIGH;
268
269 when RX_BIT_HIGH =>
270 sclk_r <= '1';
271 rx_reg(bit_cnt) <= sdata; -- Muestreamos SDA
272 if bit_cnt = 0 then
273 -- Byte completo recibido
274 data_rd <= rx_reg(7 downto 1) & sdata; -- incluimos ultimo bit
275 -- Master envia NACK (ultimo byte) ? indica fin de lectura
276 state <= MACK_SCL_LOW;
277 else
278 bit_cnt <= bit_cnt - 1;
279 state <= RX_BIT_LOW;
280 end if;
281
282 -- ---------------------------------------------------------
283 -- Master envia NACK tras recibir el byte (fin de lectura)
284 -- ---------------------------------------------------------
285 when MACK_SCL_LOW =>
286 sclk_r <= '0';
287 sdata_oe <= '1';
288 sdata_out <= '1'; -- NACK = SDA en alto
289 state <= MACK_SCL_HIGH;
290
291 when MACK_SCL_HIGH =>
292 sclk_r <= '1';
293 state <= STOP_SCL_LOW;
294
295 -- ---------------------------------------------------------
296 -- Condicion STOP: SDA sube con SCL alto
297 -- ---------------------------------------------------------
298 when STOP_SCL_LOW =>
299 sclk_r <= '0';
300 sdata_oe <= '1';
301 sdata_out <= '0';
302 state <= STOP_SDA_LOW;
303
304 when STOP_SDA_LOW =>
305 sclk_r <= '1'; -- SCL sube primero
306 state <= STOP_SCL_HIGH;
307
308 when STOP_SCL_HIGH =>
309 sdata_out <= '1'; -- SDA sube ? STOP
310 state <= FINISHED;
311
312 -- ---------------------------------------------------------
313 when FINISHED =>
314 done <= '1';
315 busy <= '0';
316 state <= IDLE;
317
318 -- ---------------------------------------------------------
319 when ERROR_STATE =>
320 error <= '1';
321 busy <= '0';
322 sclk_r <= '1';
323 sdata_out <= '1';
324 sdata_oe <= '1';
325 -- Generamos STOP para liberar el bus antes de ir a IDLE
326 state <= STOP_SCL_LOW;
327
328 when others =>
329 state <= IDLE;
330
331 end case;
332 end if;
333 end if;
334 end process p_fsm;
335
336end Behavioral;
