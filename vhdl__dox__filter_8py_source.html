105def _build_comment_map(source: str) -> dict[int, str]:
106 """Devuelve un dict {nº_línea_1based: comentario_doxygen} del fuente."""
107 comment_map = {}
108 for i, line in enumerate(source.splitlines(), start=1):
109 m = re.search(r'--!\s*(.+)$', line)
110 if m:
111 text = m.group(1).strip()
112 # Ignorar líneas que son bloques \startuml, \brief, etc.
113 if not text.startswith('\\'):
114 comment_map[i] = text
115 return comment_map

117def _clean(name: str) -> str:
118 """Elimina el '?' que pyGHDL añade a nombres no resueltos."""
119 return str(name).replace('?', '').strip()
120
121def _format_condition(cond) -> str:
122 """Reconstruye una condición EqualExpression con formato VHDL correcto."""
123 try:
124 from pyGHDL.dom.Expression import EqualExpression
125 from pyGHDL.dom.Literal import CharacterLiteral
126 if isinstance(cond, EqualExpression):
127 left = _clean(str(cond.LeftOperand))
128 right = cond.RightOperand
129 if isinstance(right, CharacterLiteral):
130 return f"{left} = '{right.Value}'"
131 return f"{left} = {_clean(str(right))}"
132 except Exception:
133 pass
134 return _clean(str(cond))

140class GhdlParser:
141 """
142 Parsea un fichero VHDL con pyGHDL y devuelve un VhdlEntity.
143 Si pyGHDL no está disponible lanza ImportError.

146 def parse(self, path: Path, source: str) -> Optional[VhdlEntity]:
147 from pyGHDL.dom.NonStandard import Document
148 from pyGHDL.dom.Object import Signal as GhdlSignal, Constant as GhdlConstant
149 from pyGHDL.dom.Type import EnumeratedType
150 from pyGHDL.dom.Concurrent import ProcessStatement
151 from pyGHDL.dom.Sequential import (
152 CaseStatement,
153 SequentialSimpleSignalAssignment,
154 IfStatement,
155 )
156 # pyGHDL necesita UTF-8 — si el fichero es UTF-16 lo convertimos
157 raw = path.read_bytes()
158 tmp = None
159 parse_path = path
160 if raw[:2] in (b'\xff\xfe', b'\xfe\xff'):
161 tmp = tempfile.NamedTemporaryFile(suffix='.vhd', delete=False,
162 mode='w', encoding='utf-8')
163 tmp.write(source)
164 tmp.close()
165 parse_path = Path(tmp.name)
166 try:
167 doc = Document(path)
168 except Exception as e:
169 print(f"[vhdl_dox_filter] pyGHDL error en {path}: {e}", file=sys.stderr)
170 return None
171
172 if not doc.Entities:
173 return None
174
175 comment_map = _build_comment_map(source)
176
177 # Tomamos la primera entidad del fichero
178 entity_name, ghdl_entity = next(iter(doc.Entities.items()))
179 entity = VhdlEntity(name=_clean(entity_name))
180
181 # --- Genéricos ---
182 for g in ghdl_entity.GenericItems:
183 for ident in g.Identifiers:
184 line = getattr(g, 'Position', None)
185 line_no = line.Line if line else None
186 entity.generics.append(Generic(
187 name=_clean(ident),
188 type_str=_clean(g.Subtype),
189 default=_clean(g.DefaultExpression) if g.DefaultExpression else None,
190 comment=comment_map.get(line_no),
191 ))
192
193 # --- Puertos ---
194 for p in ghdl_entity.PortItems:
195 for ident in p.Identifiers:
196 line = getattr(p, 'Position', None)
197 line_no = line.Line if line else None
198 entity.ports.append(Port(
199 name=_clean(ident),
200 direction=_clean(p.Mode).lower(),
201 type_str=_clean(p.Subtype),
202 default=_clean(p.DefaultExpression) if p.DefaultExpression else None,
203 comment=comment_map.get(line_no),
204 ))
205
206 # --- Arquitectura ---
207 if not doc.Architectures:
208 return entity
209
210 _, archs = next(iter(doc.Architectures.items()))
211 _, arch = next(iter(archs.items()))
212
213 # Señales y constantes en la sección declarativa
214 for item in arch.DeclaredItems:
215 line = getattr(item, 'Position', None)
216 line_no = line.Line if line else None
217
218 if isinstance(item, GhdlSignal):
219 for ident in item.Identifiers:
220 entity.signals.append(Signal(
221 name=_clean(ident),
222 type_str=_clean(item.Subtype),
223 default=_clean(item.DefaultExpression) if item.DefaultExpression else None,
224 comment=comment_map.get(line_no),
225 ))
226
227 elif isinstance(item, GhdlConstant):
228 for ident in item.Identifiers:
229 entity.constants.append(Constant(
230 name=_clean(ident),
231 type_str=_clean(item.Subtype),
232 value=_clean(item.DefaultExpression) if item.DefaultExpression else '?',
233 comment=comment_map.get(line_no),
234 ))
235
236 elif isinstance(item, EnumeratedType):
237 entity.enum_types.append(EnumType(
238 name=_clean(item.Identifier),
239 states=[_clean(lit) for lit in item.Literals],
240 ))
241
242 # --- FSM: buscar CaseStatement sobre señal de estado ---
243 enum_names = {t.name.lower() for t in entity.enum_types}
244 state_signals = {
245 s.name.lower() for s in entity.signals
246 if s.type_str.lower() in enum_names and 'next' not in s.name.lower()
247 }
248
249 if state_signals:
250 known_states = {s.lower() for t in entity.enum_types for s in t.states}
251 entity.fsm_transitions = self._extract_fsm(
252 arch.Statements, state_signals, known_states,
253 CaseStatement, SequentialSimpleSignalAssignment, IfStatement
254 )
255 entity.fsm_states = getattr(self, '_last_fsm_states', [])
256 entity.fsm_pointers = getattr(self, '_last_fsm_pointers', [])
257
258 return entity

261 def _extract_fsm(self, statements, state_signals, known_states,
262 CaseStatement, SequentialSimpleSignalAssignment, IfStatement):
263 transitions = []
264 fsm_states_dict = {}
265 pointer_uses = {}
266
267
268 for stmt in statements:
269 proc_stmts = getattr(stmt, 'Statements', [])
270 case_stmts = self._find_case_statements(proc_stmts, CaseStatement, IfStatement)
271
272 for case_stmt in case_stmts:
273 case_expr = _clean(str(case_stmt._expression)).lower()
274 if not any(sig in case_expr for sig in state_signals):
275 continue
276
277 for branch in case_stmt.Cases:
278 choices = getattr(branch, 'Choices', [])
279 if not choices:
280 continue
281 from_state = _clean(str(choices[0].Expression))
282 if from_state.upper() in ('OTHERS', 'NULL', ''):
283 continue
284
285 state_actions = []
286 fsm_states_dict[from_state] = state_actions
287
288 self._extract_branch_transitions(
289 branch.Statements, from_state, state_signals,
290 transitions, CaseStatement,
291 SequentialSimpleSignalAssignment, IfStatement,
292 known_states=known_states,
293 state_actions=state_actions,
294 pointer_uses=pointer_uses
295 )
296 # Recoger valores posibles de cada puntero
297 pointer_values = {}
298 if pointer_uses:
299 pointer_values = self._collect_pointer_values(
300 statements,
301 {k.lower() for k in pointer_uses.keys()},
302 known_states,
303 SequentialSimpleSignalAssignment, IfStatement
304 )
305
306 self._last_fsm_pointers = [
307 FsmPointer(
308 name=ptr_name,
309 used_from=used_from,
310 possible_values=pointer_values.get(ptr_name.lower(), [])
311 )
312 for ptr_name, used_from in pointer_uses.items()
313 ]
314 self._last_fsm_states = [
315 FsmState(name=name, actions=actions)
316 for name, actions in fsm_states_dict.items()
317 ]
318
319 # Deduplicar transiciones
320 seen = set()
321 unique_transitions = []
322 for tr in transitions:
323 key = (tr.from_state, tr.to_state, tr.condition)
324 if key not in seen:
325 seen.add(key)
326 unique_transitions.append(tr)
327 transitions = unique_transitions
328
329 return transitions

331 def _find_case_statements(self, statements, CaseStatement, IfStatement):
332 """Busca CaseStatements recursivamente en cualquier nivel de anidamiento."""
333 result = []
334 for s in statements:
335 if isinstance(s, CaseStatement):
336 result.append(s)
337 elif isinstance(s, IfStatement):
338 for branch in [s.IfBranch] + list(getattr(s, 'ElsIfBranches', []) or []):
339 result.extend(self._find_case_statements(
340 getattr(branch, 'Statements', []), CaseStatement, IfStatement
341 ))
342 if s.ElseBranch:
343 result.extend(self._find_case_statements(
344 getattr(s.ElseBranch, 'Statements', []), CaseStatement, IfStatement
345 ))
346 return result

348 def _extract_branch_transitions(self, statements, from_state, state_signals,
349 transitions, CaseStatement,
350 SequentialSimpleSignalAssignment, IfStatement,
351 condition=None, known_states=None,
352 state_actions=None, pointer_uses=None):
353 for bs in statements:
354 if isinstance(bs, IfStatement):
355 cond = _format_condition(bs.IfBranch._condition)
356 self._extract_branch_transitions(
357 bs.IfBranch.Statements, from_state, state_signals,
358 transitions, CaseStatement,
359 SequentialSimpleSignalAssignment, IfStatement,
360 condition=cond, known_states=known_states,
361 state_actions=state_actions, pointer_uses=pointer_uses
362 )
363 # ElsIf branches
364 for elsif in getattr(bs, 'ElsIfBranches', []) or []:
365 elsif_cond = _format_condition(elsif._condition)
366 self._extract_branch_transitions(
367 getattr(elsif, 'Statements', []), from_state, state_signals,
368 transitions, CaseStatement,
369 SequentialSimpleSignalAssignment, IfStatement,
370 condition=elsif_cond, known_states=known_states,
371 state_actions=state_actions, pointer_uses=pointer_uses
372 )
373 # Else branch
374 if bs.ElseBranch:
375 self._extract_branch_transitions(
376 bs.ElseBranch.Statements, from_state, state_signals,
377 transitions, CaseStatement,
378 SequentialSimpleSignalAssignment, IfStatement,
379 condition=f'not ({cond})', known_states=known_states,
380 state_actions=state_actions, pointer_uses=pointer_uses
381 )
382
383 elif isinstance(bs, SequentialSimpleSignalAssignment):
384 target = _clean(str(bs.Target)).lower()
385 if any(sig in target for sig in state_signals):
386 for wave in bs.Waveform:
387 to_state = _clean(str(wave.Expression))
388 if to_state.upper() in ('OTHERS', 'NULL', ''):
389 continue
390 if known_states and to_state.lower() not in known_states:
391 if pointer_uses is not None:
392 key = to_state
393 if key not in pointer_uses:
394 pointer_uses[key] = []
395 if from_state not in pointer_uses[key]:
396 pointer_uses[key].append(from_state)
397 continue
398 transitions.append(FsmTransition(
399 from_state=from_state,
400 to_state=to_state,
401 condition=condition,
402 ))
403 else:
404 if state_actions is not None and from_state not in [None, '']:
405 for wave in bs.Waveform:
406 value = _clean(str(wave.Expression))
407 action = f'{_clean(str(bs.Target))} <= {value}'
408 if action not in state_actions:
409 state_actions.append(action)

411 def _collect_pointer_values(self, statements, pointer_names, known_states,
412 SequentialSimpleSignalAssignment, IfStatement):
413 """
414 Recorre todo el código buscando asignaciones a señales puntero
415 para recoger todos sus posibles valores.
416 """
417 values = {p: [] for p in pointer_names}
418
419 def search(stmts):
420 for s in stmts:
421 if isinstance(s, SequentialSimpleSignalAssignment):
422 target = _clean(str(s.Target)).lower()
423 if target in pointer_names:
424 for wave in s.Waveform:
425 val = _clean(str(wave.Expression))
426 if known_states and val.lower() in known_states:
427 if val not in values[target]:
428 values[target].append(val)
429 elif isinstance(s, IfStatement):
430 search(getattr(s.IfBranch, 'Statements', []))
431 if s.ElseBranch:
432 search(getattr(s.ElseBranch, 'Statements', []))
433
434 for stmt in statements:
435 search(getattr(stmt, 'Statements', []))
436
437 return values

754class Injector:
755 _RE_ENTITY_START = re.compile(r'^(\s*entity\s+\w+\s+is)', re.IGNORECASE | re.MULTILINE)

757 def inject(self, source: str, block: str) -> str:
758 m = self._RE_ENTITY_START.search(source)
759 if not m:
760 return source
761
762 insert_pos = m.start()
763
764 # Buscar el último \brief antes de "entity X is"
765 brief_re = re.compile(r'(--!\s*\\brief[^\n]*\n)', re.IGNORECASE)
766 brief_match = None
767 for bm in brief_re.finditer(source[:insert_pos]):
768 brief_match = bm
769
770 if not brief_match:
771 return source[:insert_pos] + block + source[insert_pos:]
772
773 after_brief = brief_match.end()
774 pre_entity = source[after_brief:insert_pos]
775
776 # Construir \details conservando todo el contenido existente
777 # (texto, WaveDrom, UML manual) y añadiendo las tablas al final
778 details = '--! \\details\n'
779 details += pre_entity
780 details += block
781
782 return source[:after_brief] + details + source[insert_pos:]
