25class HierarchyParser:
26 """
27 Extrae la jerarquía de instanciaciones VHDL usando regex.
28
29 Se usa regex en lugar de pyGHDL porque este script se ejecuta antes
30 de Doxygen y necesita ser robusto con cualquier sintaxis VHDL,
31 incluyendo ficheros con errores de compilación o packages.
32 """
33
34 # Detecta declaración de entidad: entity nombre is
35 _RE_ENTITY = re.compile(
36 r'\bentity\s+(\w+)\s+is\b',
37 re.IGNORECASE
38 )
39
40 # Detecta instanciación directa: label : entity work.nombre
41 _RE_ENTITY_INST = re.compile(
42 r':\s*entity\s+\w+\.(\w+)\s*(?:generic|port|\n)',
43 re.IGNORECASE
44 )
45
46 # Palabras reservadas VHDL para evitar falsos positivos en instanciaciones
47 _VHDL_KEYWORDS = {
48 'process', 'begin', 'end', 'if', 'else', 'elsif', 'case', 'when',
49 'loop', 'for', 'while', 'wait', 'signal', 'variable', 'constant',
50 'port', 'generic', 'map', 'architecture', 'package', 'library',
51 'use', 'with', 'select', 'generate', 'block', 'component',
52 'function', 'procedure', 'return', 'type', 'subtype', 'record',
53 'array', 'of', 'in', 'out', 'inout', 'buffer', 'linkage',
54 'std_logic', 'std_logic_vector', 'integer', 'natural', 'boolean',
55 'bit', 'bit_vector', 'string', 'real', 'time', 'others', 'null',
56 'open', 'true', 'false', 'rising_edge', 'falling_edge',
57 'to_integer', 'to_unsigned', 'to_signed', 'resize', 'conv_integer',
58 }
59
60 # Detecta instanciación por componente: label : ComponentName port map
61 _RE_COMPONENT_INST = re.compile(
62 r'(\w+)\s*:\s*(\w+)\s*(?:generic\s+map|port\s+map)',
63 re.IGNORECASE
64 )
65
66 _RE_ARCH = re.compile(
67 r'\barchitecture\s+(\w+)\s+of\s+(\w+)\s+is\b',
68 re.IGNORECASE
69 )
70
71 def parse_file(self, path: Path) -> list[VhdlModule]:
72 """
73 Parsea un fichero VHDL y devuelve los módulos (entidades) encontrados
74 con sus instanciaciones.
75 """
76 try:
77 source = path.read_text(encoding='utf-8', errors='replace')
78 except Exception as e:
79 print(f"[generate_hierarchy] Error leyendo {path}: {e}", file=sys.stderr)
80 return []
81
82 # Eliminar comentarios de línea para evitar falsos positivos
83 source_clean = re.sub(r'--[^\n]*', '', source)
84
85 modules = []
86
87 for entity_match in self._RE_ENTITY.finditer(source_clean):
88 entity_name = entity_match.group(1)
89
90 # Ignorar "end entity nombre" — solo queremos la declaración
91 before = source_clean[max(0, entity_match.start()-10):entity_match.start()]
92 if re.search(r'\bend\s*$', before, re.IGNORECASE):
93 continue
94
95 module = VhdlModule(name=entity_name, path=path)
96
97 # Detectar arquitectura
98 arch_match = self._RE_ARCH.search(source_clean)
99 if arch_match:
100 module.architecture = arch_match.group(1)
101
102 rest = source_clean[entity_match.start():]
103
104 # 1. Instanciaciones directas: entity work.X o entity lib.X
105 for inst_match in self._RE_ENTITY_INST.finditer(rest):
106 inst_name = inst_match.group(1)
107 if inst_name.lower() not in self._VHDL_KEYWORDS:
108 if inst_name not in module.instantiates:
109 module.instantiates.append(inst_name)
110
111 # 2. Instanciaciones por componente: label : ComponentName port map
112 for comp_match in self._RE_COMPONENT_INST.finditer(rest):
113 comp_name = comp_match.group(2)
114 if comp_name.lower() not in self._VHDL_KEYWORDS:
115 if comp_name not in module.instantiates:
116 module.instantiates.append(comp_name)
117
118 modules.append(module)
119
120 return modules

127class HierarchyDiagramGenerator:
128 """
129 Genera el diagrama PlantUML de jerarquía a partir de los módulos extraídos.
130 """
131
132 def generate(self, modules: list[VhdlModule]) -> str:
133 """
134 Genera el string PlantUML del diagrama de jerarquía.
135 Las flechas representan instanciaciones: A --> B significa A instancia B.
136 Los módulos raíz (no instanciados por nadie) se marcan en verde.
137 """
138 # Detectar módulos raíz (no instanciados por ningún otro módulo)
139 all_instantiated = set()
140 for m in modules:
141 for inst in m.instantiates:
142 all_instantiated.add(inst.lower())
143
144 root_modules = {m.name for m in modules if m.name.lower() not in all_instantiated}
145
146 lines = [
147 '@startuml',
148 'skinparam componentStyle rectangle',
149 'skinparam defaultFontName Helvetica',
150 'skinparam component {',
151 ' BackgroundColor #E8F4FD',
152 ' BorderColor #2E86AB',
153 ' FontColor #1a1a1a',
154 '}',
155 'skinparam arrow {',
156 ' Color #2E86AB',
157 ' FontColor #1a1a1a',
158 '}',
159 'left to right direction',
160 '',
161 ]
162
163 # Declarar módulos raíz con color verde
164 for m in modules:
165 if m.name in root_modules:
166 lines.append(f'component "{m.name}" as {m.name} #D4EDDA')
167
168 lines.append('')
169
170 # Generar relaciones de instanciación
171 added = set()
172 for m in modules:
173 for inst_name in m.instantiates:
174 key = (m.name, inst_name)
175 if key not in added:
176 lines.append(f'{m.name} --> {inst_name}')
177 added.add(key)
178
179 lines.append('')
180 lines.append('@enduml')
181
182 return '\n'.join(lines)

189class DoxFileGenerator:
190 """Genera el fichero .dox con la página Doxygen del diagrama de jerarquía."""
191
192 def generate(self, plantuml_diagram: str, modules: list[VhdlModule]) -> str:
193 """
194 Genera el contenido del fichero .dox con:
195 - Página Doxygen con título y descripción
196 - Diagrama PlantUML de jerarquía
197 - Tabla de módulos con sus ficheros fuente e instanciaciones
198 """
199 # Tabla de módulos
200 table_lines = [
201 '| Module | File | Instantiates |',
202 '|--------|------|--------------|',
203 ]
204 for m in sorted(modules, key=lambda x: x.name.lower()):
205 inst_str = ', '.join(f'`{i}`' for i in m.instantiates) if m.instantiates else '—'
206 table_lines.append(f'| `{m.name}` | `{m.path.name}` | {inst_str} |')
207
208 table = '\n'.join(table_lines)
209
210 dox = f'''/*!
211\\page page_hierarchy DESIGN HIERARCHY
212
213\\brief Jerarquía de instanciaciones del proyecto VHDL.
214
215Este diagrama muestra la jerarquía de instanciaciones entre las entidades
216del proyecto. Los módulos en verde son los módulos raíz (top-level),
217es decir, los que no son instanciados por ningún otro módulo.
218
219\\par Hierarchy Diagram
220\\startuml
221{plantuml_diagram}
222\\enduml
223
224\\par Module List
225{table}
226
227*/
228'''
229 return dox
