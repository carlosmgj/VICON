66class GhdlReportParser:
67 """
68 Parsea la salida de GHDL -a.
69 Formato: fichero:linea:col: (error|warning): mensaje
70 """
71 _RE = re.compile(
72 r'^(.+?):(\d+):(\d+):\s*(error|warning|note):\s*(.+)$',
73 re.IGNORECASE
74 )
75
76 # Patrones de errores que son warnings en realidad (IPs externos, primitivas Vivado, etc.)
77 RECLASSIFY_AS_WARNING = [
78 r'not found in library',
79 r'cannot find entity',
80 r'unit ".+" not found',
81 ]
82
83 def parse(self, content: str) -> ToolReport:
84 report = ToolReport(tool='GHDL', raw_content=content, available=True)
85 for line in content.splitlines():
86 m = self._RE.match(line.strip())
87 if m:
88 severity = m.group(4).lower()
89 if severity == 'note':
90 severity = 'info'
91
92 # Reclasificar errores conocidos como warnings
93 if severity == 'error':
94 message = m.group(5).strip()
95 for pattern in self.RECLASSIFY_AS_WARNINGRECLASSIFY_AS_WARNING:
96 if re.search(pattern, message, re.IGNORECASE):
97 severity = 'warning'
98 break
99
100 report.issues.append(ReportIssue(
101 file=Path(m.group(1)).name,
102 line=int(m.group(2)),
103 column=int(m.group(3)),
104 severity=severity,
105 rule='',
106 message=m.group(5).strip(),
107 ))
108 return report

111class VsgReportParser:
112 """

114 Formato: ERROR: fichero(linea)regla -- mensaje
115 """
116 _RE = re.compile(
117 r'^(ERROR|WARNING):\s*(.+?)\‍((\d+)\‍)(\w+)\s*--\s*(.+)$',
118 re.IGNORECASE
119 )
120
121 def parse(self, content: str) -> ToolReport:
122 report = ToolReport(tool='VSG', raw_content=content, available=True)
123 for line in content.splitlines():
124 m = self._RE.match(line.strip())
125 if m:
126 report.issues.append(ReportIssue(

180 def _tool_section(self, report: ToolReport) -> str:
181 """Genera la sección de una herramienta."""
182 n_errors = len(report.errors)
183 n_warnings = len(report.warnings)
184 n_total = len(report.issues)
185
186 # Badge de estado
187 if n_errors == 0 and n_warnings == 0:
188 status = '✅ PASSED'
189 elif n_errors == 0:
190 status = f'⚠️ {n_warnings} warnings'
191 else:
192 status = f'❌ {n_errors} errors, {n_warnings} warnings'
193
194 lines = [
195 f'\\section sec_report_{report.tool.lower()} {report.tool} — {status}',
196 '',
197 ]
198
199 # Resumen
200 lines += [
201 '\\par Summary',
202 f'| Tool | Total | Errors | Warnings |',
203 f'|------|-------|--------|----------|',
204 f'| `{report.tool}` | {n_total} | {n_errors} | {n_warnings} |',
205 '',
206 ]
207
208 if not report.issues:
209 lines.append('No issues found.')
210 lines.append('')
211 return '\n'.join(lines)
212
213 # Agrupar por fichero
214 by_file: dict[str, list[ReportIssue]] = {}
215 for issue in report.issues:
216 by_file.setdefault(issue.file, []).append(issue)
217
218 lines.append('\\par Issues by file')
219 lines.append('')
220
221 for fname, issues in sorted(by_file.items()):
222 n_e = sum(1 for i in issues if i.severity == 'error')
223 n_w = sum(1 for i in issues if i.severity == 'warning')
224 lines.append(f'\\subsection sec_{report.tool.lower()}_{self._safe_id(fname)} {fname} ({n_e} errors, {n_w} warnings)')
225 lines.append('')
226 lines.append('| Line | Col | Severity | Rule | Message |')
227 lines.append('|------|-----|----------|------|---------|')
228 for issue in sorted(issues, key=lambda x: x.line):
229 sev_badge = '🔴' if issue.severity == 'error' else '🟡'
230 rule_str = f'`{issue.rule}`' if issue.rule else '—'
231 base = Path(issue.file).stem.lower().replace('_', '__')
232 url = self._vhd_to_dox_url(issue.file, issue.line)
233 lines.append(
234 f'| {issue.line} | {issue.column} | {sev_badge} {issue.severity} | {rule_str} | {issue.message} |'
235 )
236 lines.append('')
237
238 return '\n'.join(lines)
