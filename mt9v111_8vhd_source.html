74
75 signal regs_core : reg_map_t := (
76 16#00# => x"823A", -- Chip Version (ID real del MT9V111)
77 16#0D# => x"0008", -- Reset/Misc Control (valor por defecto)
78 others => x"CACA"

86begin
87
88 -- Generador de Pixel Clock independiente
89 clk_int <= not clk_int after PIX_PERIOD/2;
90
91 -----------------------------------------------------------
92 -- PROCESO 2: Respondedor I2C Activo (Slave)
93 -----------------------------------------------------------
94 i2c_active_respond: process
95 variable v_reg_addr : integer range 0 to 255;
96 variable v_data_16 : std_logic_vector(15 downto 0);
97 variable v_addr_dev : std_logic_vector(7 downto 0);
98 variable write_n_read : std_logic;
99 begin
100 sda <= 'Z';
101
102 -- 1. START
103 wait until falling_edge(sda) and scl = '1';
104
105 -- 2. Recibir Dirección Dispositivo (8 bits)
106 for i in 7 downto 0 loop
107 wait until rising_edge(scl);
108 v_addr_dev(i) := sda;
109 end loop;
110
111 if v_addr_dev(7 downto 1) = I2C_ADDR then
112 -- ACK 1 (Dirección Dispositivo)
113 wait until falling_edge(scl);
114 sda <= '0';
115 wait until falling_edge(scl);
116 sda <= 'Z';
117 write_n_read := v_addr_dev(0);
118
119 if (write_n_read = '0') then
120 log_to_file("reporte_final_1.txt", "COMANDO DE ESCRITURA", false);
121 else
122 log_to_file("reporte_final_1.txt", "COMANDO DE LECTURA", false);
123 end if;
124
125
126 -- 3. Recibir Dirección de Registro (8 bits)
127 for i in 7 downto 0 loop
128 wait until rising_edge(scl);
129 v_data_16(i+8) := sda; -- Usamos v_data_16 temporalmente para ahorrar variables
130 end loop;
131 v_reg_addr := to_integer(unsigned(v_data_16(15 downto 8)));
132
133
134 -- ACK 2 (Dirección Registro)
135 wait until falling_edge(scl);
136 sda <= '0';
137 wait until falling_edge(scl);
138 sda <= 'Z';
139
140 -- 4. Recibir Dato BYTE ALTO (8 bits)
141 for i in 7 downto 0 loop
142 wait until rising_edge(scl);
143 v_data_16(i+8) := sda;
144 end loop;
145
146 -- ACK 3 (Byte Alto)
147 wait until falling_edge(scl);
148 sda <= '0';
149 wait until falling_edge(scl);
150 sda <= 'Z';
151
152 -- 5. Recibir Dato BYTE BAJO (8 bits)
153 for i in 7 downto 0 loop
154 wait until rising_edge(scl);
155 v_data_16(i) := sda;
156 end loop;
157
158 -- ACK 4 (Byte Bajo)
159 wait until falling_edge(scl);
160 sda <= '0';
161 wait until falling_edge(scl);
162 sda <= 'Z';
163
164 -- 6. GUARDAR EN EL MAPA (Suponiendo que estamos en la página 0)
165 regs_core(v_reg_addr) <= v_data_16;
166 log_to_file("reporte_final_1.txt", "I2C Agente: Registro 0x" &
167 int_to_hex_str(v_reg_addr, 2) & " escrito con valor 0x" & int_to_hex_str(to_integer(unsigned(v_data_16)),4), false);
168
169 end if;
170
171 -- 7. STOP
172 wait until rising_edge(sda) and scl = '1';
173 log_to_file("reporte_final_1.txt", "I2C Agente: STOP detectado", false);
174 end process;
175
176end Behavioral;
