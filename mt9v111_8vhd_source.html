44 -- Registro base para lectura: el master primero escribe addr_reg en modo

57 ---------------------------------------------------------------------------
58 -- Proceso I2C esclavo continuo.
59 --
60 -- Secuencia WRITE:
61 -- S + ADDR_WR + ACK + REG_ADDR + ACK + DATA_H + ACK + DATA_L + ACK + [más regs] + P
62 --
63 -- Secuencia READ (el master hace dos transacciones seguidas):
64 -- 1ª: S + ADDR_WR + ACK + REG_ADDR + ACK (establece el registro a leer)
65 -- 2ª: Sr + ADDR_RD + ACK + [agente envía DATA_H + master ACK + DATA_L + master ACK/NACK] + P
66 ---------------------------------------------------------------------------
67 i2c_slave : process
68 variable v_addr_byte : std_logic_vector(7 downto 0);
69 variable v_rw : std_logic;
70 variable v_reg_addr : integer range 0 to 255;

74 variable v_data_16 : std_logic_vector(15 downto 0);
75 variable v_stop_seen : boolean;
76 variable v_mack : std_logic;
77 variable v_send_byte : std_logic_vector(7 downto 0);

79 sda <= 'Z';
80
81 -- Loop exterior: una iteración = una transacción I2C completa

83
84 -- ----------------------------------------------------------------
85 -- Esperar START o Repeated START:
86 -- SDA baja mientras SCL está alto
87 -- ----------------------------------------------------------------
88 wait until falling_edge(sda) and scl = '1';
89 log_to_file("reporte_final_1.txt", "I2C Agente: START detectado", false);
90
91 -- ----------------------------------------------------------------
92 -- Recibir byte de dirección (7 bits addr + 1 bit R/W)
93 -- ----------------------------------------------------------------

95 wait until rising_edge(scl);
96 v_addr_byte(i) := sda;
97 end loop;

99
100 -- ----------------------------------------------------------------
101 -- Solo responder si la dirección es la nuestra
102 -- ----------------------------------------------------------------
103 if v_addr_byte(7 downto 1) = I2C_ADDR then
104
105 -- ACK de dirección
106 wait until falling_edge(scl);
107 sda <= '0';
108 wait until falling_edge(scl);
109 sda <= 'Z';
110
111 -- ============================================================
112 -- MODO WRITE (R/W='0')
113 -- ============================================================
114 if v_rw = '0' then
115
116 log_to_file("reporte_final_1.txt",
117 "I2C Agente: Direccion reconocida, modo WRITE", false);

120 for i in 7 downto 0 loop
121 wait until rising_edge(scl);
122 v_addr_byte(i) := sda;
123 end loop;
124 v_reg_addr := to_integer(unsigned(v_addr_byte));
125 v_reg_inc := v_reg_addr;
126 read_reg_addr <= v_reg_addr; -- guardar para posible lectura posterior
127
128 -- ACK de dirección de registro
129 wait until falling_edge(scl);
130 sda <= '0';
131 wait until falling_edge(scl);
132 sda <= 'Z';
133
134 log_to_file("reporte_final_1.txt", "I2C Agente: Registro base 0x" &
135 int_to_hex_str(v_reg_addr, 2), false);
136
137 -- Bucle de recepción de datos con auto-increment.
138 -- Termina al detectar STOP (SDA sube con SCL alto).
139 v_stop_seen := false;
140
141 while not v_stop_seen loop
142
143 wait on scl, sda;
144
145 if scl = '1' and sda = '1' then
146 -- Condición STOP
147 v_stop_seen := true;
148 log_to_file("reporte_final_1.txt",
149 "I2C Agente: STOP detectado, fin escritura", false);
150
151 elsif scl = '0' then
152 -- SCL bajó: recibir DATA_H
153 for i in 7 downto 0 loop
154 wait until rising_edge(scl);
155 v_data_h(i) := sda;
156 end loop;
157
158 -- ACK DATA_H
159 wait until falling_edge(scl);
160 sda <= '0';
161 wait until falling_edge(scl);
162 sda <= 'Z';
163
164 -- Recibir DATA_L

166 wait until rising_edge(scl);
167 v_data_l(i) := sda;
168 end loop;
169
170 -- ACK DATA_L
171 wait until falling_edge(scl);
172 sda <= '0';
173 wait until falling_edge(scl);
174 sda <= 'Z';
175
176 -- Guardar con auto-increment
177 v_data_16 := v_data_h & v_data_l;
178 regs_core(v_reg_inc) <= v_data_16;
179
180 log_to_file("reporte_final_1.txt",
181 "I2C Agente: Reg 0x" & int_to_hex_str(v_reg_inc, 2) &
182 " = 0x" & int_to_hex_str(to_integer(unsigned(v_data_16)), 4), false);
183
184 if v_reg_inc = 255 then
185 v_reg_inc := 0;
186 else
187 v_reg_inc := v_reg_inc + 1;
188 end if;
189
190 end if;
191
192 end loop;
193
194 -- ============================================================
195 -- MODO READ (R/W='1')
196 -- El master ya nos indicó el registro en la transacción write
197 -- previa (guardado en read_reg_addr).
198 -- El agente conduce SDA con los bits del registro solicitado.
199 -- El master controla SCL en todo momento.
200 -- ============================================================
201 else
202
203 log_to_file("reporte_final_1.txt",
204 "I2C Agente: Direccion reconocida, modo READ desde reg 0x" &
205 int_to_hex_str(read_reg_addr, 2), false);
206
207 v_reg_inc := read_reg_addr;
208 v_stop_seen := false;
209
210 while not v_stop_seen loop
211
212 -- ------------------------------------------------
213 -- Enviar DATA_H del registro actual
214 -- ------------------------------------------------
215 v_send_byte := regs_core(v_reg_inc)(15 downto 8);
216
217 for i in 7 downto 0 loop
218 -- Esperar flanco de bajada de SCL para cambiar SDA
219 wait until falling_edge(scl);
220 sda <= v_send_byte(i);
221 end loop;
222
223 -- Soltar SDA para que el master envíe ACK
224 wait until falling_edge(scl);
225 sda <= 'Z';
226
227 -- Leer ACK del master (flanco de subida de SCL)
228 wait until rising_edge(scl);
229 v_mack := sda;

231 if v_mack = '1' then
232 -- NACK: el master no quiere más datos
233 v_stop_seen := true;
234 log_to_file("reporte_final_1.txt",
235 "I2C Agente: NACK recibido tras DATA_H, fin lectura", false);
236 else
237 -- ------------------------------------------------
238 -- Enviar DATA_L del mismo registro
239 -- ------------------------------------------------
240 v_send_byte := regs_core(v_reg_inc)(7 downto 0);
241
242 for i in 7 downto 0 loop
243 wait until falling_edge(scl);
244 sda <= v_send_byte(i);
245 end loop;
246
247 -- Soltar SDA para ACK del master
248 wait until falling_edge(scl);
249 sda <= 'Z';
250
251 -- Leer ACK/NACK del master
252 wait until rising_edge(scl);
253 v_mack := sda;
254
255 log_to_file("reporte_final_1.txt",
256 "I2C Agente: Reg 0x" & int_to_hex_str(v_reg_inc, 2) &
257 " leido: 0x" & int_to_hex_str(
258 to_integer(unsigned(regs_core(v_reg_inc))), 4), false);
259
260 if v_mack = '1' then
261 -- NACK: fin de lectura
262 v_stop_seen := true;
263 log_to_file("reporte_final_1.txt",
264 "I2C Agente: NACK recibido, fin lectura multiple", false);
265 else
266 -- ACK: auto-increment y continuar
267 if v_reg_inc = 255 then
268 v_reg_inc := 0;
269 else
270 v_reg_inc := v_reg_inc + 1;
271 end if;
272 end if;
273
274 end if;
275
276 end loop;
277
278 -- Esperar STOP del master
279 wait until rising_edge(sda) and scl = '1';
280 log_to_file("reporte_final_1.txt",
281 "I2C Agente: STOP detectado, fin lectura", false);
282
283 end if; -- v_rw
284
285 else
286 -- Dirección no reconocida
287 log_to_file("reporte_final_1.txt",
288 "I2C Agente: Direccion no reconocida, ignorando", false);
289 wait until rising_edge(sda) and scl = '1';
290
291 end if; -- I2C_ADDR
292
293 end loop; -- loop exterior
294
295 end process i2c_slave;
296
297end Behavioral;
