128 p_fsm : process(clk)
129 begin
130 if rising_edge(clk) then
131 if reset = '1' then
132 state <= IDLE;
133 sclk_r <= '1';
134 sdata_out <= '1';
135 sdata_oe <= '0';
136 busy <= '0';
137 done <= '0';
138 error <= '0';
139 bit_cnt <= 7;
140 byte_cnt <= 0;
141 in_read <= '0';
142 last_data <= '0';
143 data_rd <= (others => '0');
144
145 elsif tick = '1' then
146
147 -- Defecto: pulsos de un ciclo
148 done <= '0';
149 error <= '0';
150
151 case state is
152
153 -- ---------------------------------------------------------
154 when IDLE =>
155 busy <= '0';
156 sclk_r <= '1';
157 sdata_out <= '1';
158 sdata_oe <= '1';
159 in_read <= '0';
160 if start = '1' then
161 -- Latch de parámetros
162 rw_r <= rw;
163 nb_r <= num_bytes;
164 addr_r <= addr_dev;
165 areg_r <= addr_reg;
166 dwr_r <= data_wr;
167 busy <= '1';
168 byte_cnt <= 0;
169 state <= START_SDA_LOW;
170 end if;
171
172 -- ---------------------------------------------------------
173 -- Condición START: SDA baja con SCL alto
174 -- ---------------------------------------------------------
175 when START_SDA_LOW =>
176 sdata_out <= '0'; -- SDA baja ? START
177 state <= START_SCL_LOW;
178
179 when START_SCL_LOW =>
180 sclk_r <= '0'; -- SCL baja para empezar a transmitir
181 -- Cargamos primer byte: ADDR + R/W
182 -- Si in_read='1' estamos en repeated start ? ADDR+R
183 -- Si in_read='0' ? ADDR+W
184 if in_read = '1' then
185 shift_reg <= addr_r & '1';
186 else
187 shift_reg <= addr_r & '0';
188 end if;
189 bit_cnt <= 7;
190 state <= TX_BIT_LOW;
191
192 -- ---------------------------------------------------------
193 -- Transmisión de un byte (master ? esclavo)
194 -- ---------------------------------------------------------
195 when TX_BIT_LOW =>
196 sclk_r <= '0';
197 sdata_oe <= '1';
198 sdata_out <= shift_reg(bit_cnt);
199 state <= TX_BIT_HIGH;
200
201 when TX_BIT_HIGH =>
202 sclk_r <= '1';
203 if bit_cnt = 0 then
204 state <= ACK_SCL_LOW;
205 else
206 bit_cnt <= bit_cnt - 1;
207 state <= TX_BIT_LOW;
208 end if;
209
210 -- ---------------------------------------------------------
211 -- Espera ACK del esclavo
212 -- ---------------------------------------------------------
213 when ACK_SCL_LOW =>
214 sclk_r <= '0';
215 sdata_oe <= '0'; -- Soltamos SDA para que esclavo ponga ACK
216 state <= ACK_SCL_HIGH;
217
218 when ACK_SCL_HIGH =>
219 sclk_r <= '1';
220 if sdata = '1' then
221 -- NACK ? abortar
222 state <= ERROR_STATE;
223 else
224 -- ACK OK ? decidir siguiente acción según byte_cnt
225 if in_read = '1' then
226 -- Acabamos de enviar ADDR+R, empezamos a leer
227 bit_cnt <= 7;
228 rx_reg <= (others => '0');
229 state <= RX_BIT_LOW;
230 elsif byte_cnt = 0 then
231 -- Fin de ADDR+W ? enviar registro
232 shift_reg <= areg_r;
233 bit_cnt <= 7;
234 byte_cnt <= 1;
235 state <= TX_BIT_LOW;
236 elsif byte_cnt = 1 then
237 -- Fin de REG
238 if rw_r = '1' then
239 -- Repeated START para lectura
240 in_read <= '1';
241 sdata_out <= '1';
242 sdata_oe <= '1';
243 state <= START_SDA_LOW;
244 else
245 -- Escritura: enviar bytes de datos
246 shift_reg <= dwr_r(31 downto 24);
247 bit_cnt <= 7;
248 byte_cnt <= 2;
249 state <= TX_BIT_LOW;
250 end if;
251 elsif byte_cnt <= nb_r then
252 -- Enviando bytes de datos (byte_cnt va de 2 a nb_r+1)
253 -- byte_cnt=2 ? dwr_r(23..16), 3?(15..8), 4?(7..0)
254 case byte_cnt is
255 when 2 => shift_reg <= dwr_r(23 downto 16);
256 when 3 => shift_reg <= dwr_r(15 downto 8);
257 when others => shift_reg <= dwr_r(7 downto 0);
258 end case;
259 bit_cnt <= 7;
260 byte_cnt <= byte_cnt + 1;
261 state <= TX_BIT_LOW;
262 else
263 -- Todos los bytes enviados ? STOP
264 state <= STOP_SCL_LOW;
265 end if;
266 end if;
267
268 -- ---------------------------------------------------------
269 -- Recepción de un byte (esclavo ? master)
270 -- ---------------------------------------------------------
271 when RX_BIT_LOW =>
272 sclk_r <= '0';
273 sdata_oe <= '0'; -- Master en alta impedancia
274 state <= RX_BIT_HIGH;
275
276 when RX_BIT_HIGH =>
277 sclk_r <= '1';
278 rx_reg(bit_cnt) <= sdata; -- Muestreamos SDA
279 if bit_cnt = 0 then
280 -- Byte completo recibido
281 data_rd <= rx_reg(7 downto 1) & sdata; -- incluimos último bit
282 -- Master envía NACK (último byte) ? indica fin de lectura
283 state <= MACK_SCL_LOW;
284 else
285 bit_cnt <= bit_cnt - 1;
286 state <= RX_BIT_LOW;
287 end if;
288
289 -- ---------------------------------------------------------
290 -- Master envía NACK tras recibir el byte (fin de lectura)
291 -- ---------------------------------------------------------
292 when MACK_SCL_LOW =>
293 sclk_r <= '0';
294 sdata_oe <= '1';
295 sdata_out <= '1'; -- NACK = SDA en alto
296 state <= MACK_SCL_HIGH;
297
298 when MACK_SCL_HIGH =>
299 sclk_r <= '1';
300 state <= STOP_SCL_LOW;
301
302 -- ---------------------------------------------------------
303 -- Condición STOP: SDA sube con SCL alto
304 -- ---------------------------------------------------------
305 when STOP_SCL_LOW =>
306 sclk_r <= '0';
307 sdata_oe <= '1';
308 sdata_out <= '0';
309 state <= STOP_SDA_LOW;
310
311 when STOP_SDA_LOW =>
312 sclk_r <= '1'; -- SCL sube primero
313 state <= STOP_SCL_HIGH;
314
315 when STOP_SCL_HIGH =>
316 sdata_out <= '1'; -- SDA sube ? STOP
317 state <= FINISHED;
318
319 -- ---------------------------------------------------------
320 when FINISHED =>
321 done <= '1';
322 busy <= '0';
323 state <= IDLE;
324
325 -- ---------------------------------------------------------
326 when ERROR_STATE =>
327 error <= '1';
328 busy <= '0';
329 sclk_r <= '1';
330 sdata_out <= '1';
331 sdata_oe <= '1';
332 -- Generamos STOP para liberar el bus antes de ir a IDLE
333 state <= STOP_SCL_LOW;
334
335 when others =>
336 state <= IDLE;
337
338 end case;
339 end if;
340 end if;
341 end process p_fsm;
