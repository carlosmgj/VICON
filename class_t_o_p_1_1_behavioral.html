133 process(clk)
134 begin
135 if rising_edge(clk) then
136 if reset = '1' then
137 state <= ST_IDLE;
138 i2c_rw <= '0';
139 i2c_start <= '0';
140 i2c_wr_push <= '0';
141 i2c_rd_pop <= '0';
142 i2c_num_regs <= 1;
143 i2c_addr_reg <= (others => '0');
144 i2c_wr_data <= (others => '0');
145 fill_cnt <= 0;
146 rd_cnt <= 0;
147 done <= '0';
148 else
149 -- Pulsos de un ciclo por defecto
150 i2c_start <= '0';
151 i2c_wr_push <= '0';
152 i2c_rd_pop <= '0';
153
154 case state is
155
156 -----------------------------------------------------------
157 when ST_IDLE =>
158 done <= '0';
159 fill_cnt <= 0;
160 rd_cnt <= 0;
161 state <= ST_WR_FILL_FIFO;
162
163 -----------------------------------------------------------
164 -- Cargar datos en WR FIFO antes de lanzar la escritura.
165 -- Se comprueban dos condiciones: FIFO no llena y aún
166 -- quedan datos por meter.
167 -----------------------------------------------------------
168 when ST_WR_FILL_FIFO =>
169 if fill_cnt = NUM_REGS_WR then
170 -- FIFO cargada → lanzar escritura
171 fill_cnt <= 0;
172 state <= ST_WR_START;
173 elsif i2c_wr_full = '0' then
174 i2c_wr_data <= WR_DATA(fill_cnt);
175 i2c_wr_push <= '1';
176 fill_cnt <= fill_cnt + 1;
177 end if;
178 -- Si FIFO llena y aún quedan datos: esperar (raro con
179 -- FIFO_DEPTH >= NUM_REGS_WR, pero seguro)
180
181 -----------------------------------------------------------
182 when ST_WR_START =>
183 if i2c_busy = '0' then
184 i2c_rw <= '0'; -- Write
185 i2c_addr_reg <= x"04"; -- Registro inicial
186 i2c_num_regs <= NUM_REGS_WR;
187 i2c_start <= '1';
188 state <= ST_WR_WAIT;
189 end if;
190
191 -----------------------------------------------------------
192 when ST_WR_WAIT =>
193 if i2c_error = '1' then
194 state <= ST_ERROR;
195 elsif i2c_done = '1' then
196 state <= ST_RD_START;
197 end if;
198
199 -----------------------------------------------------------
200 when ST_RD_START =>
201 -- El controlador bloquea si RD FIFO no está vacía,
202 -- pero aquí sabemos que lo está (acabamos de arrancar)
203 if i2c_busy = '0' and i2c_rd_empty = '1' then
204 i2c_rw <= '1'; -- Read
205 i2c_addr_reg <= x"04";
206 i2c_num_regs <= NUM_REGS_RD;
207 i2c_start <= '1';
208 state <= ST_RD_WAIT;
209 end if;
210
211 -----------------------------------------------------------
212 when ST_RD_WAIT =>
213 if i2c_error = '1' then
214 state <= ST_ERROR;
215 elsif i2c_done = '1' then
216 rd_cnt <= 0;
217 state <= ST_RD_DRAIN;
218 end if;
219
220 -----------------------------------------------------------
221 -- Vaciar RD FIFO y guardar en buffer interno.
222 -- Cada ciclo en que rd_fifo_empty='0' se hace pop.
223 -----------------------------------------------------------
224 when ST_RD_DRAIN =>
225 if i2c_rd_empty = '0' and rd_cnt < NUM_REGS_RD then
226 i2c_rd_pop <= '1';
227 rd_buf(rd_cnt) <= i2c_rd_data;
228 rd_cnt <= rd_cnt + 1;
229 elsif rd_cnt = NUM_REGS_RD then
230 state <= ST_FINISH;
231 end if;
232
233 -----------------------------------------------------------
234 when ST_FINISH =>
235 done <= '1';
236 state <= ST_FINISH; -- Mantener done='1'
237 -- Aquí rd_buf(0) y rd_buf(1) tienen los valores leídos
238
239 -----------------------------------------------------------
240 when ST_ERROR =>
241 done <= '0';
242 state <= ST_ERROR;
243 -- Señal de error visible en i2c_error del controlador.
244 -- Añadir lógica de recuperación si es necesario.
245
246 when others =>
247 state <= ST_IDLE;
248
249 end case;
250 end if;
251 end if;
252 end process;
